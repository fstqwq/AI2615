int n;
vector <int> g[N];
bool visited[N];
void dfs (int u) {
    visited[u] = true;
    for (auto v : g[u]) {
        if (! visited[v]) {
            dfs (v);
        }
    }
}
void traverse_the_graph () {
    // clear flag
    fill (visited + 1, visited + n + 1, false);
    // enumerated possible start
    // subject to change if there are extra requirements:
    // e.g., if every point is reachable by 1, dfs(1) is sufficient
    for (int i = 1; i <= n; i++) {
        if (! visited[i]) {
            dfs (i);
        }
    }
}

void bfs (int start) {
    // clear flag
    fill (visited + 1, visited + n + 1, false);
    queue <int> q;
    q.push(start);
    visited[start] = true;
    while (! q.empty()) {
        int u = q.front();
        q.pop();
        for (auto v : g[u]) {
            if (! visited[v]) {
                visited[v] = true;
                q.push(v);
            }
        }
    }
}

DFS 序与 BFS 序

如果你关心 DFS 或者 BFS 中顺序的信息，你可以额外记录下来。

比如，如果你关心 DFS 序，那你可以在访问到一个顶点的时候记录 order[step ++] = u;。我们以 Kosaraju 算法中需要用到的 Finishing Time 为例：

int order[N], step = 1;
void dfs (int u) {
    visited[u] = true;
    for (auto v : g[u]) {
        if (! visited[v]) {
            dfs (v);
        }
    }
    order[step ++] = u;
}
void traverse_the_graph () {
    fill (visited + 1, visited + n + 1, false);
    for (int i = 1; i <= n; i++) {
        if (! visited[i]) {
            dfs (i);
        }
    }
    // order[1...n] now contains the finishing time
}

int q[N], head, tail;
void bfs () {
    head = 1, tail = 0;
    q[++ tail] = start;
    while (head <= tail) {
        int u = q[head ++];
        for (auto v : g[u]) {
            if (! visited[v]) {
                visited[v] = true;
                q[++ tail] = v;
            }
        }
    }
    // q[1...tail] now contains the bfs order
}
